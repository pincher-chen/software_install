# 2.Compile
```
$module load intel-compilers/15.0.1
$module load MPI/Intel/MPICH/3.1-icc15-dyn  
$module load fftw/3.3.4-double-fma-icc15
$export BLAS_LIBS="-L/HOME/intel/composer_xe_2015.1.133/mkl/lib/intel64 -lmkl_intel_lp64  -lmkl_sequential -lmkl_core"
$export LAPACK_LIBS="-L/HOME/intel/composer_xe_2015.1.133/mkl/lib/intel64 -lmkl_intel_lp64  -lmkl_sequential -lmkl_core"
$export SCALAPACK_LIBS="-L/HOME/intel/composer_xe_2015.1.133/mkl/lib/intel64 -lmkl_scalapack_lp64 -lmkl_blacs_intelmpi_lp64"
$export FFT_LIBS="-lfftw3 -lfftw3_mpi"

$./configure --prefix="/WORK/app/Quantum_Espresso/6.1_MPI" --enable-parallel --enable-shared --with-scalapack=intel FC=mpif90 MPIF90=mpif90 F77=mpif90 CC=mpicc

$make all
$make install
```
